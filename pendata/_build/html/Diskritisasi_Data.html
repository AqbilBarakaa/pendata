Hitung Information Gain tiap fitur dan simpan threshold terbaiknya.

# 4. Hitung Information Gain & simpan threshold terbaik tiap fitur
for feature in features:
    unique_values = sorted(df[feature].unique())  # Contoh: ['A', 'B', 'C', 'D']

    best_ig = -1
    best_threshold = None
    best_entropies = {}

    for threshold in unique_values:
        df[f'{feature}_binary'] = df[feature].apply(lambda x: 1 if x > threshold else 0)

        X = df[[f'{feature}_binary']]
        y = df['ClassEncoded']

        clf = DecisionTreeClassifier(criterion='entropy', max_depth=1, random_state=42)
        clf.fit(X, y)

        tree = clf.tree_
        if tree.node_count < 3:
            continue

        n_total = tree.n_node_samples[0]
        left_count = tree.n_node_samples[1]
        right_count = tree.n_node_samples[2]

        e_root = tree.impurity[0]
        e_left = tree.impurity[1]
        e_right = tree.impurity[2]

        avg_entropy = (left_count / n_total) * e_left + (right_count / n_total) * e_right
        info_gain = e_root - avg_entropy

        if info_gain > best_ig:
            best_ig = info_gain
            best_threshold = threshold
            best_entropies = {
                'Entropy Awal': round(e_root, 4),
                'Entropy Kiri': round(e_left, 4),
                'Entropy Kanan': round(e_right, 4),
                'Entropy Rata2': round(avg_entropy, 4),
                'Information Gain': round(info_gain, 4)
            }

    results[feature] = {
        'Threshold Terbaik': best_threshold,
        **best_entropies
    }
    thresholds[feature] = best_threshold
    df[f'{feature}_binary'] = df[feature].apply(lambda x: 1 if x > best_threshold else 0)
